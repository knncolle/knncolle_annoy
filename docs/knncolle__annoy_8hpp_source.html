84 void search(Index_ i, Index_ k, std::vector<std::pair<Index_, Float_> >& output) {
85 Index_ kp1 = k + 1;
86 my_indices.clear();
87 my_indices.reserve(kp1);
88 my_distances.clear();
89 my_distances.reserve(kp1);
90 my_parent->my_index.get_nns_by_item(i, kp1, get_search_k(kp1), &my_indices, &my_distances); // +1, as it forgets to discard 'self'.
91
92 output.clear();
93 output.reserve(k);
94 bool self_found = false;
95 for (size_t x = 0, end = my_indices.size(); x < end; ++x) {
96 if (!self_found && my_indices[x] == i) {
97 self_found = true;
98 } else {
99 output.emplace_back(my_indices[x], my_distances[x]);
100 }
101 }
102
103 // Just in case we're full of ties at duplicate points, such that 'c'
104 // is not in the set. Note that, if self_found=false, we must have at
105 // least 'K+2' points for 'c' to not be detected as its own neighbor.
106 // Thus there is no need to worry whether we are popping off a non-'c'
107 // element and then returning fewer elements than expected.
108 if (!self_found) {
109 output.pop_back();
110 }
111 }
